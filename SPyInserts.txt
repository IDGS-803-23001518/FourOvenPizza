CREATE PROCEDURE sp_gestion_usuarios(
    IN p_accion      VARCHAR(20),
    IN p_idUsuario   INT,
    IN p_idRol       INT,
    IN p_nombre      VARCHAR(100),
    IN p_usuario     VARCHAR(50),
    IN p_contrasenia VARCHAR(255),
    IN p_estatus     TINYINT,
    IN p_email       VARCHAR(150),
    IN p_ip          VARCHAR(45),
    IN p_ejecutadoPor INT,
    OUT p_resultado  VARCHAR(255),
    OUT p_idGenerado INT
)
BEGIN
    DECLARE v_nombre_ejecutor VARCHAR(100) DEFAULT 'Sistema';

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SET p_resultado = 'ERROR: Ocurrió un error en la operación';
        SET p_idGenerado = 0;
        INSERT INTO bitacora_sistema (nivel, modulo, mensaje, detalles, fecha, ip, usuarioId)
        VALUES ('ERROR', 'Usuarios',
                CONCAT('Error en sp_gestion_usuarios acción: ', p_accion),
                CONCAT('Usuario afectado: ', IFNULL(p_usuario, ''), ' | Ejecutado por ID: ', IFNULL(p_ejecutadoPor, 0)),
                NOW(), p_ip, p_ejecutadoPor);
    END;

    IF p_ejecutadoPor IS NOT NULL THEN
        SELECT nombre INTO v_nombre_ejecutor FROM usuarios WHERE idUsuario = p_ejecutadoPor;
    END IF;

    START TRANSACTION;

    IF p_accion = 'INSERT' THEN
        IF EXISTS (SELECT 1 FROM usuarios WHERE usuario = p_usuario) THEN
            SET p_resultado = 'ERROR: El nombre de usuario ya existe';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSEIF EXISTS (SELECT 1 FROM usuarios WHERE email = p_email) THEN
            SET p_resultado = 'ERROR: El correo ya está registrado';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSEIF NOT EXISTS (SELECT 1 FROM roles WHERE idRol = p_idRol AND estatus = 1) THEN
            SET p_resultado = 'ERROR: El rol seleccionado no existe o está inactivo';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSE
            INSERT INTO usuarios (idRol, nombre, usuario, email, contrasenia, estatus, fechaCreacion)
            VALUES (p_idRol, p_nombre, p_usuario, p_email, p_contrasenia, 1, NOW());
            SET p_idGenerado = LAST_INSERT_ID();
            SET p_resultado = 'SUCCESS: Usuario registrado correctamente';

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES (p_ejecutadoPor, v_nombre_ejecutor, 'Usuarios', 'CREAR',
                    'usuario', CONCAT('ID:', p_idGenerado, ' | ', p_usuario), NOW(), p_ip);
            COMMIT;
        END IF;

    ELSEIF p_accion = 'UPDATE' THEN
        IF NOT EXISTS (SELECT 1 FROM usuarios WHERE idUsuario = p_idUsuario) THEN
            SET p_resultado = 'ERROR: El usuario no existe';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSEIF EXISTS (SELECT 1 FROM usuarios WHERE usuario = p_usuario AND idUsuario != p_idUsuario) THEN
            SET p_resultado = 'ERROR: El nombre de usuario ya está en uso';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSEIF EXISTS (SELECT 1 FROM usuarios WHERE email = p_email AND idUsuario != p_idUsuario) THEN
            SET p_resultado = 'ERROR: El correo ya está en uso por otro usuario';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSEIF NOT EXISTS (SELECT 1 FROM roles WHERE idRol = p_idRol AND estatus = 1) THEN
            SET p_resultado = 'ERROR: El rol no existe o está inactivo';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSE
            UPDATE usuarios
            SET idRol=p_idRol, nombre=p_nombre, usuario=p_usuario,
                email=p_email, estatus=p_estatus
            WHERE idUsuario = p_idUsuario;

            IF p_contrasenia IS NOT NULL AND p_contrasenia != '' THEN
                UPDATE usuarios SET contrasenia=p_contrasenia WHERE idUsuario=p_idUsuario;
            END IF;

            SET p_idGenerado = p_idUsuario;
            SET p_resultado = 'SUCCESS: Usuario actualizado correctamente';

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES (p_ejecutadoPor, v_nombre_ejecutor, 'Usuarios', 'EDITAR',
                    'usuario', CONCAT('ID:', p_idUsuario, ' | ', p_usuario), NOW(), p_ip);
            COMMIT;
        END IF;

    ELSEIF p_accion = 'CHANGE_STATUS' THEN
        IF NOT EXISTS (SELECT 1 FROM usuarios WHERE idUsuario = p_idUsuario) THEN
            SET p_resultado = 'ERROR: El usuario no existe';
            SET p_idGenerado = 0;
            ROLLBACK;
        ELSE
            UPDATE usuarios SET estatus=p_estatus WHERE idUsuario=p_idUsuario;
            SET p_idGenerado = p_idUsuario;
            SET p_resultado = IF(p_estatus=1,
                'SUCCESS: Usuario activado correctamente',
                'SUCCESS: Usuario desactivado correctamente');

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES (p_ejecutadoPor, v_nombre_ejecutor, 'Usuarios',
                    IF(p_estatus=1, 'ACTIVAR', 'DESACTIVAR'),
                    'usuario', CONCAT('ID:', p_idUsuario), NOW(), p_ip);
            COMMIT;
        END IF;

    ELSE
        SET p_resultado = 'ERROR: Acción no válida';
        SET p_idGenerado = 0;
        ROLLBACK;
    END IF;
END$$

CREATE PROCEDURE sp_cambiar_estatus_rol(
    IN  p_idRol        INT,
    IN  p_estatus      TINYINT,
    IN  p_ejecutadoPor INT,
    IN  p_ip           VARCHAR(45),
    OUT p_resultado    VARCHAR(100)
)
BEGIN
    DECLARE v_nombre_ejecutor VARCHAR(100) DEFAULT 'Sistema';
    DECLARE v_nombre_rol      VARCHAR(50)  DEFAULT '';

    IF p_ejecutadoPor IS NOT NULL THEN
        SELECT nombre INTO v_nombre_ejecutor FROM usuarios WHERE idUsuario = p_ejecutadoPor;
    END IF;

    IF NOT EXISTS (SELECT 1 FROM roles WHERE idRol = p_idRol) THEN
        SET p_resultado = 'ERROR: Rol no encontrado';

    ELSEIF p_estatus = 0 AND (SELECT COUNT(*) FROM roles WHERE estatus = 1) <= 1 THEN
        SET p_resultado = 'ERROR: Debe existir al menos un rol activo en el sistema';

    ELSE
        SELECT nombre INTO v_nombre_rol FROM roles WHERE idRol = p_idRol;
        UPDATE roles SET estatus = p_estatus WHERE idRol = p_idRol;

        INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
        VALUES (p_ejecutadoPor, v_nombre_ejecutor, 'Roles',
                IF(p_estatus=1, 'ACTIVAR', 'DESACTIVAR'),
                'rol', CONCAT('ID:', p_idRol, ' | ', v_nombre_rol), NOW(), p_ip);

        SET p_resultado = 'OK';
    END IF;
END$$

DELIMITER $$

DROP PROCEDURE IF EXISTS sp_gestion_materiasprimas$$
CREATE PROCEDURE sp_gestion_materiasprimas(
    IN p_accion         VARCHAR(20),
    IN p_idMateriaP     INT,
    IN p_nombre         VARCHAR(100),
    IN p_tipo           VARCHAR(50),
    IN p_idCategoria    INT,
    IN p_stock          DECIMAL(10,2),
    IN p_stockMinimo    DECIMAL(10,2),
    IN p_estatus        TINYINT,
    IN p_ip             VARCHAR(45),
    IN p_ejecutadoPor   INT,
    OUT p_resultado     VARCHAR(255),
    OUT p_idGenerado    INT
)
BEGIN

    DECLARE v_nombre_ejecutor VARCHAR(100) DEFAULT 'Sistema';
    DECLARE v_nombre_mp VARCHAR(100) DEFAULT '';

    -- Manejo de errores
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SET p_resultado = 'ERROR: Ocurrió un error en la operación';
        SET p_idGenerado = 0;

        INSERT INTO bitacora_sistema 
        (nivel, modulo, mensaje, detalles, fecha, ip, usuarioId)
        VALUES (
            'ERROR',
            'MateriasPrimas',
            CONCAT('Error en sp_gestion_materiasprimas acción: ', p_accion),
            CONCAT('Materia: ', IFNULL(p_nombre,'')),
            NOW(),
            p_ip,
            p_ejecutadoPor
        );
    END;

    -- Obtener ejecutor
    IF p_ejecutadoPor IS NOT NULL THEN
        SELECT nombre INTO v_nombre_ejecutor 
        FROM usuarios 
        WHERE idUsuario = p_ejecutadoPor;
    END IF;

    START TRANSACTION;

    /* =========================================
       INSERT
    ========================================= */

    IF p_accion = 'INSERT' THEN

        IF EXISTS (SELECT 1 FROM materiasprimas WHERE nombre = p_nombre) THEN
            SET p_resultado = 'ERROR: Ya existe una materia prima con ese nombre';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSEIF NOT EXISTS (SELECT 1 FROM categorias 
                           WHERE idCategoria = p_idCategoria 
                           AND estatus = 1) THEN
            SET p_resultado = 'ERROR: Categoría inválida';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSEIF p_stock < 0 THEN
            SET p_resultado = 'ERROR: Stock negativo no permitido';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            INSERT INTO materiasprimas
            (nombre, tipo, idCategoria, stock, stockMinimo, estatus)
            VALUES
            (p_nombre, p_tipo, p_idCategoria, p_stock, p_stockMinimo, 1);

            SET p_idGenerado = LAST_INSERT_ID();
            SET p_resultado = 'SUCCESS: Materia prima registrada correctamente';

            INSERT INTO bitacora_eventos
            (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES
            (
                p_ejecutadoPor,
                v_nombre_ejecutor,
                'MateriasPrimas',
                'CREAR',
                'materiaprima',
                CONCAT('ID:', p_idGenerado,' | ',p_nombre),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;


    /* =========================================
       UPDATE
    ========================================= */

    ELSEIF p_accion = 'UPDATE' THEN

        IF NOT EXISTS (SELECT 1 FROM materiasprimas WHERE idMateriaP = p_idMateriaP) THEN
            SET p_resultado = 'ERROR: Materia prima no existe';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSEIF EXISTS (
            SELECT 1 
            FROM materiasprimas 
            WHERE nombre = p_nombre 
            AND idMateriaP != p_idMateriaP
        ) THEN

            SET p_resultado = 'ERROR: Ya existe otra materia prima con ese nombre';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            UPDATE materiasprimas
            SET
                nombre = p_nombre,
                tipo = p_tipo,
                idCategoria = p_idCategoria,
                stock = p_stock,
                stockMinimo = p_stockMinimo,
                estatus = IFNULL(p_estatus, estatus)
            WHERE idMateriaP = p_idMateriaP;

            SET p_idGenerado = p_idMateriaP;
            SET p_resultado = 'SUCCESS: Materia prima actualizada correctamente';

            INSERT INTO bitacora_eventos
            (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES
            (
                p_ejecutadoPor,
                v_nombre_ejecutor,
                'MateriasPrimas',
                'EDITAR',
                'materiaprima',
                CONCAT('ID:', p_idMateriaP,' | ',p_nombre),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;


    /* =========================================
       CHANGE_STATUS
    ========================================= */

    ELSEIF p_accion = 'CHANGE_STATUS' THEN

        IF NOT EXISTS (SELECT 1 FROM materiasprimas WHERE idMateriaP = p_idMateriaP) THEN
            SET p_resultado = 'ERROR: Materia prima no existe';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            UPDATE materiasprimas
            SET estatus = p_estatus
            WHERE idMateriaP = p_idMateriaP;

            SET p_idGenerado = p_idMateriaP;

            SET p_resultado = IF(
                p_estatus = 1,
                'SUCCESS: Materia prima activada',
                'SUCCESS: Materia prima desactivada'
            );

            INSERT INTO bitacora_eventos
            (usuarioId, nombreUsuario, modulo, accion, referencial, referencia, fecha, ip)
            VALUES
            (
                p_ejecutadoPor,
                v_nombre_ejecutor,
                'MateriasPrimas',
                IF(p_estatus=1,'ACTIVAR','DESACTIVAR'),
                'materiaprima',
                CONCAT('ID:',p_idMateriaP),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;


    /* =========================================
       DELETE
    ========================================= */

    ELSEIF p_accion = 'DELETE' THEN

    IF NOT EXISTS (
        SELECT 1 
        FROM materiasprimas 
        WHERE idMateriaP = p_idMateriaP
    ) THEN

        SET p_resultado = 'ERROR: Materia prima no existe';
        SET p_idGenerado = 0;
        ROLLBACK;

    ELSE

        SELECT nombre 
        INTO v_nombre_mp 
        FROM materiasprimas 
        WHERE idMateriaP = p_idMateriaP;

        UPDATE materiasprimas 
        SET estatus = 0
        WHERE idMateriaP = p_idMateriaP;

        SET p_idGenerado = p_idMateriaP;
        SET p_resultado = 'SUCCESS: Materia prima desactivada correctamente';

        INSERT INTO bitacora_eventos
        (
            usuarioId,
            nombreUsuario,
            modulo,
            accion,
            referencial,
            referencia,
            fecha,
            ip
        )
        VALUES
        (
            p_ejecutadoPor,
            v_nombre_ejecutor,
            'MateriasPrimas',
            'ELIMINAR',
            'materiaprima',
            CONCAT('ID:',p_idMateriaP,' | ',v_nombre_mp),
            NOW(),
            p_ip
        );

        COMMIT;
    END IF;

    ELSE

        SET p_resultado = 'ERROR: Acción no válida';
        SET p_idGenerado = 0;
        ROLLBACK;

    END IF;

END$$

CREATE PROCEDURE sp_gestion_unidadesmedida(
    IN p_accion VARCHAR(10),
    IN p_idUnidad INT,
    IN p_nombre VARCHAR(50),
    IN p_tipo VARCHAR(50),
    IN p_equivalente DECIMAL(10,2),
    IN p_estatus TINYINT,
    IN p_ip VARCHAR(50),
    IN p_usuario INT,
    OUT p_resultado VARCHAR(255),
    OUT p_idGenerado INT
)

BEGIN

DECLARE v_nombre VARCHAR(50);
DECLARE v_nombre_usuario VARCHAR(100) DEFAULT 'Sistema';

IF p_usuario IS NOT NULL THEN
    SELECT nombre INTO v_nombre_usuario FROM usuarios WHERE idUsuario = p_usuario;
END IF;

START TRANSACTION;

IF p_accion = 'INSERT' THEN

    IF EXISTS (SELECT 1 FROM unidadesmedida WHERE nombre = p_nombre) THEN
    
        SET p_resultado = 'ERROR: La unidad ya existe';
        SET p_idGenerado = 0;
        ROLLBACK;

    ELSE
    
        INSERT INTO unidadesmedida
        (nombre, tipo, equivalente, estatus)
        VALUES
        (p_nombre, p_tipo, p_equivalente, 1);

        SET p_idGenerado = LAST_INSERT_ID();

        SET p_resultado = 'SUCCESS: Unidad registrada correctamente';

        INSERT INTO bitacora_eventos
        (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
        VALUES
        (p_usuario, v_nombre_usuario, 'UnidadesMedida', 'Registrar', p_nombre, NOW(), p_ip);

        COMMIT;

    END IF;


ELSEIF p_accion = 'UPDATE' THEN

    IF NOT EXISTS (SELECT 1 FROM unidadesmedida WHERE idUnidadM = p_idUnidad) THEN
    
        SET p_resultado = 'ERROR: Unidad no existe';
        SET p_idGenerado = 0;
        ROLLBACK;

    ELSE

        UPDATE unidadesmedida
        SET
        nombre = p_nombre,
        tipo = p_tipo,
        equivalente = p_equivalente,
        estatus = IFNULL(p_estatus, estatus)
        WHERE idUnidadM = p_idUnidad;

        SET p_resultado = 'SUCCESS: Unidad actualizada';
        SET p_idGenerado = p_idUnidad;

        INSERT INTO bitacora_eventos
        (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
        VALUES
        (p_usuario, v_nombre_usuario, 'UnidadesMedida', 'Editar', p_nombre, NOW(), p_ip);

        COMMIT;

    END IF;


ELSEIF p_accion = 'DELETE' THEN

    IF NOT EXISTS (SELECT 1 FROM unidadesmedida WHERE idUnidadM = p_idUnidad) THEN

        SET p_resultado = 'ERROR: Unidad no existe';
        SET p_idGenerado = 0;
        ROLLBACK;

    ELSE

        UPDATE unidadesmedida
        SET estatus = 0
        WHERE idUnidadM = p_idUnidad;

        SET p_resultado = 'SUCCESS: Unidad desactivada';
        SET p_idGenerado = p_idUnidad;

        INSERT INTO bitacora_eventos
        (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
        VALUES
        (p_usuario, v_nombre_usuario, 'UnidadesMedida', 'Eliminar', p_idUnidad, NOW(), p_ip);

        COMMIT;

    END IF;

END IF;

END$$

ALTER TABLE productos
ADD UNIQUE KEY uk_nombre_tamano (nombre, `tamaño`);
 
DELIMITER $$
DROP PROCEDURE IF EXISTS sp_gestion_productos$$
CREATE PROCEDURE sp_gestion_productos(
    IN  p_accion     VARCHAR(10),
    IN  p_idProducto INT,
    IN  p_nombre     VARCHAR(100),
    IN  p_precio     DECIMAL(10,2),
    IN  p_tamano     VARCHAR(50),       -- sin acento, se normaliza dentro
    IN  p_stock      DECIMAL(10,2),
    IN  p_estatus    TINYINT,
    IN  p_ip         VARCHAR(50),
    IN  p_usuario    INT,
    OUT p_resultado  VARCHAR(255),
    OUT p_idGenerado INT
)
BEGIN
    DECLARE v_nombre_producto VARCHAR(100) DEFAULT '';
    DECLARE v_nombre_usuario  VARCHAR(100) DEFAULT 'Sistema';
 
    -- ── Variables normalizadas ────────────────────────
    -- Se trabaja siempre con las mismas versiones limpias
    DECLARE v_nombre VARCHAR(100);
    DECLARE v_tamano VARCHAR(50);
 
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SET p_resultado = 'ERROR: Ocurrió un error en la operación';
        SET p_idGenerado = 0;
        INSERT INTO bitacora_sistema (nivel, modulo, mensaje, detalles, fecha, ip, usuarioId)
        VALUES ('ERROR', 'Productos',
                CONCAT('Error en sp_gestion_productos acción: ', p_accion),
                CONCAT('Producto: ', IFNULL(p_nombre,''), ' | ID: ', IFNULL(p_idProducto,0)),
                NOW(), p_ip, p_usuario);
    END;
 
    -- Obtener nombre del ejecutor
    IF p_usuario IS NOT NULL THEN
        SELECT nombre INTO v_nombre_usuario FROM usuarios WHERE idUsuario = p_usuario;
    END IF;
 
    -- Normalizar entradas: sin espacios extra, en minúsculas para comparar
    SET v_nombre = LOWER(TRIM(p_nombre));
    SET v_tamano = LOWER(TRIM(p_tamano));
 
    START TRANSACTION;
 
    /* ══════════════════════════════════════
       INSERT
    ══════════════════════════════════════ */
    IF p_accion = 'INSERT' THEN
 
        -- Validar nombre no vacío
        IF v_nombre = '' THEN
            SET p_resultado = 'ERROR: El nombre del producto no puede estar vacío';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar tamaño no vacío
        ELSEIF v_tamano = '' THEN
            SET p_resultado = 'ERROR: El tamaño del producto no puede estar vacío';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar precio
        ELSEIF p_precio < 0 THEN
            SET p_resultado = 'ERROR: El precio no puede ser negativo';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar stock
        ELSEIF p_stock < 0 THEN
            SET p_resultado = 'ERROR: El stock no puede ser negativo';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar duplicado: comparación exacta normalizada
        ELSEIF EXISTS (
            SELECT 1 FROM productos
            WHERE LOWER(TRIM(nombre))   = v_nombre
              AND LOWER(TRIM(`tamaño`)) = v_tamano
        ) THEN
            SET p_resultado = 'ERROR: Ya existe un producto con ese nombre y tamaño';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        ELSE
            INSERT INTO productos (nombre, precio, `tamaño`, stock, estatus)
            VALUES (TRIM(p_nombre), p_precio, TRIM(p_tamano), p_stock, 1);
 
            SET p_idGenerado = LAST_INSERT_ID();
            SET p_resultado  = 'SUCCESS: Producto registrado correctamente';
 
            INSERT INTO bitacora_eventos
                (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES
                (p_usuario, v_nombre_usuario, 'Productos', 'Registrar',
                 CONCAT('ID:', p_idGenerado, ' | ', TRIM(p_nombre), ' | ', TRIM(p_tamano)),
                 NOW(), p_ip);
 
            COMMIT;
        END IF;
 
    /* ══════════════════════════════════════
       UPDATE
    ══════════════════════════════════════ */
    ELSEIF p_accion = 'UPDATE' THEN
 
        -- Validar que existe
        IF NOT EXISTS (SELECT 1 FROM productos WHERE idProducto = p_idProducto) THEN
            SET p_resultado = 'ERROR: El producto no existe';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar nombre no vacío
        ELSEIF v_nombre = '' THEN
            SET p_resultado = 'ERROR: El nombre del producto no puede estar vacío';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar tamaño no vacío
        ELSEIF v_tamano = '' THEN
            SET p_resultado = 'ERROR: El tamaño del producto no puede estar vacío';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar precio
        ELSEIF p_precio < 0 THEN
            SET p_resultado = 'ERROR: El precio no puede ser negativo';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar stock
        ELSEIF p_stock < 0 THEN
            SET p_resultado = 'ERROR: El stock no puede ser negativo';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        -- Validar duplicado en otro registro: comparación exacta normalizada
        ELSEIF EXISTS (
            SELECT 1 FROM productos
            WHERE LOWER(TRIM(nombre))   = v_nombre
              AND LOWER(TRIM(`tamaño`)) = v_tamano
              AND idProducto           <> p_idProducto
        ) THEN
            SET p_resultado = 'ERROR: Ya existe otro producto con ese nombre y tamaño';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        ELSE
            UPDATE productos
            SET nombre   = TRIM(p_nombre),
                precio   = p_precio,
                `tamaño` = TRIM(p_tamano),
                stock    = p_stock,
                estatus  = IFNULL(p_estatus, estatus)
            WHERE idProducto = p_idProducto;
 
            SET p_resultado  = 'SUCCESS: Producto actualizado correctamente';
            SET p_idGenerado = p_idProducto;
 
            INSERT INTO bitacora_eventos
                (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES
                (p_usuario, v_nombre_usuario, 'Productos', 'Editar',
                 CONCAT('ID:', p_idProducto, ' | ', TRIM(p_nombre), ' | ', TRIM(p_tamano)),
                 NOW(), p_ip);
 
            COMMIT;
        END IF;
 
    /* ══════════════════════════════════════
       DELETE (desactivación lógica)
    ══════════════════════════════════════ */
    ELSEIF p_accion = 'DELETE' THEN
 
        IF NOT EXISTS (SELECT 1 FROM productos WHERE idProducto = p_idProducto) THEN
            SET p_resultado = 'ERROR: El producto no existe';
            SET p_idGenerado = 0;
            ROLLBACK;
 
        ELSE
            SELECT nombre INTO v_nombre_producto
            FROM productos WHERE idProducto = p_idProducto;
 
            -- Desactivación lógica (no borrado físico)
            UPDATE productos
            SET estatus = 0
            WHERE idProducto = p_idProducto;
 
            SET p_resultado  = 'SUCCESS: Producto desactivado correctamente';
            SET p_idGenerado = p_idProducto;
 
            INSERT INTO bitacora_eventos
                (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES
                (p_usuario, v_nombre_usuario, 'Productos', 'Eliminar',
                 CONCAT('ID:', p_idProducto, ' | ', v_nombre_producto),
                 NOW(), p_ip);
 
            COMMIT;
        END IF;
 
    /* ══════════════════════════════════════
       ACCIÓN NO VÁLIDA
    ══════════════════════════════════════ */
    ELSE
        SET p_resultado = 'ERROR: Acción no válida';
        SET p_idGenerado = 0;
        ROLLBACK;
    END IF;
 
END$$

DELIMITER $$

DROP PROCEDURE IF EXISTS sp_gestion_recetas $$
CREATE PROCEDURE sp_gestion_recetas(
    IN p_accion VARCHAR(10),
    IN p_idReceta INT,
    IN p_idProducto INT,
    IN p_descripcion VARCHAR(200),
    IN p_ip VARCHAR(50),
    IN p_usuario INT,
    OUT p_resultado VARCHAR(255),
    OUT p_idGenerado INT
)
BEGIN
    DECLARE v_nombre_usuario VARCHAR(100) DEFAULT 'Sistema';

    IF p_usuario IS NOT NULL THEN
        SELECT nombre INTO v_nombre_usuario FROM usuarios WHERE idUsuario = p_usuario;
    END IF;

    START TRANSACTION;

    IF p_accion = 'INSERT' THEN

        IF NOT EXISTS (
            SELECT 1
            FROM productos
            WHERE idProducto = p_idProducto
        ) THEN

            SET p_resultado = 'ERROR: El producto no existe';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSEIF EXISTS (
            SELECT 1
            FROM recetas
            WHERE idProducto = p_idProducto
        ) THEN

            SET p_resultado = 'ERROR: El producto ya tiene una receta registrada';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            INSERT INTO recetas (idProducto, descripcion)
            VALUES (p_idProducto, p_descripcion);

            SET p_idGenerado = LAST_INSERT_ID();
            SET p_resultado = 'SUCCESS: Receta registrada correctamente';

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES (
                p_usuario,
                v_nombre_usuario,
                'Recetas',
                'Registrar',
                CONCAT('Receta ID:', p_idGenerado, ' | Producto ID:', p_idProducto),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;

    ELSEIF p_accion = 'UPDATE' THEN

        IF NOT EXISTS (
            SELECT 1
            FROM recetas
            WHERE idReceta = p_idReceta
        ) THEN

            SET p_resultado = 'ERROR: La receta no existe';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            UPDATE recetas
            SET descripcion = p_descripcion
            WHERE idReceta = p_idReceta;

            SET p_resultado = 'SUCCESS: Receta actualizada correctamente';
            SET p_idGenerado = p_idReceta;

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES (
                p_usuario,
                v_nombre_usuario,
                'Recetas',
                'Editar',
                CONCAT('Receta ID:', p_idReceta),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;

    ELSEIF p_accion = 'DELETE' THEN

        IF NOT EXISTS (
            SELECT 1
            FROM recetas
            WHERE idReceta = p_idReceta
        ) THEN

            SET p_resultado = 'ERROR: La receta no existe';
            SET p_idGenerado = 0;
            ROLLBACK;

        ELSE

            DELETE FROM detalleReceta
            WHERE idReceta = p_idReceta;

            DELETE FROM recetas
            WHERE idReceta = p_idReceta;

            SET p_resultado = 'SUCCESS: Receta eliminada correctamente';
            SET p_idGenerado = p_idReceta;

            INSERT INTO bitacora_eventos (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
            VALUES (
                p_usuario,
                v_nombre_usuario,
                'Recetas',
                'Eliminar',
                CONCAT('Receta ID:', p_idReceta),
                NOW(),
                p_ip
            );

            COMMIT;

        END IF;

    ELSE

        SET p_resultado = 'ERROR: Accion no valida';
        SET p_idGenerado = 0;
        ROLLBACK;

    END IF;

END $$

-- SP para cambiar estatus de productos (lee estatus actual internamente)
DELIMITER $$

DROP PROCEDURE IF EXISTS sp_cambiar_estatus_producto$$
CREATE PROCEDURE sp_cambiar_estatus_producto(
    IN  p_idProducto   INT,
    IN  p_ip           VARCHAR(50),
    IN  p_usuario      INT,
    OUT p_resultado    VARCHAR(255),
    OUT p_nuevoEstatus TINYINT
)
BEGIN
    DECLARE v_estatus_actual  TINYINT       DEFAULT NULL;
    DECLARE v_nombre_producto VARCHAR(100)  DEFAULT '';
    DECLARE v_nombre_usuario  VARCHAR(100)  DEFAULT 'Sistema';

    IF NOT EXISTS (SELECT 1 FROM productos WHERE idProducto = p_idProducto) THEN
        SET p_resultado = 'ERROR: Producto no encontrado';
        SET p_nuevoEstatus = 0;
    ELSE
        SELECT estatus, nombre INTO v_estatus_actual, v_nombre_producto
        FROM productos WHERE idProducto = p_idProducto;

        SET p_nuevoEstatus = IF(v_estatus_actual = 1, 0, 1);

        UPDATE productos SET estatus = p_nuevoEstatus WHERE idProducto = p_idProducto;

        IF p_usuario IS NOT NULL THEN
            SELECT nombre INTO v_nombre_usuario FROM usuarios WHERE idUsuario = p_usuario;
        END IF;

        INSERT INTO bitacora_eventos
            (usuarioId, nombreUsuario, modulo, accion, referencia, fecha, ip)
        VALUES
            (p_usuario, v_nombre_usuario, 'Productos',
             IF(p_nuevoEstatus = 1, 'ACTIVAR', 'DESACTIVAR'),
             CONCAT('ID:', p_idProducto, ' | ', v_nombre_producto),
             NOW(), p_ip);

        SET p_resultado = IF(p_nuevoEstatus = 1,
            'SUCCESS: Producto activado correctamente',
            'SUCCESS: Producto desactivado correctamente');
    END IF;
END$$

BEGIN

        idMateriaP,
        nombre,
        stock,

      AND stockMinimo > 0

-- SP para obtener alertas de stock de una receta especifica
DELIMITER $$

DROP PROCEDURE IF EXISTS sp_obtener_alertas_stock_receta$$
CREATE PROCEDURE sp_obtener_alertas_stock_receta(
    IN p_idReceta INT
)
BEGIN
    SELECT
        mp.nombre    AS materia_prima,
        mp.stock     AS stock,
        mp.stockMinimo AS stock_minimo,
        dr.cantidad  AS requerido,
        CASE
            WHEN mp.stock < dr.cantidad THEN 'insuficiente'
            WHEN mp.stockMinimo > 0 AND mp.stock <= mp.stockMinimo THEN 'bajo'
            ELSE 'ok'
        END AS tipo_alerta
    FROM detalleReceta dr
    INNER JOIN materiasPrimas mp ON mp.idMateriaP = dr.idMateriaP
    WHERE dr.idReceta = p_idReceta
    ORDER BY mp.nombre ASC;
END$$

-- SP para listar productos con filtros
DELIMITER $$

DROP PROCEDURE IF EXISTS sp_listar_productos$$
CREATE PROCEDURE sp_listar_productos(
    IN p_nombre         VARCHAR(100),
    IN p_precio_inicio  DECIMAL(10,2),
    IN p_precio_fin     DECIMAL(10,2),
    IN p_estatus        VARCHAR(10),
    IN p_tamano         VARCHAR(50),
    IN p_estatus_receta VARCHAR(10)
)
BEGIN
    SELECT
        p.idProducto,
        p.nombre,
        p.precio,
        p.`tamaño` AS tamano,
        p.stock,
        p.estatus,
        CASE WHEN EXISTS (SELECT 1 FROM recetas r WHERE r.idProducto = p.idProducto)
             THEN 1 ELSE 0 END AS receta_activa
    FROM productos p
    WHERE (p_nombre IS NULL OR p.nombre LIKE CONCAT('%', p_nombre, '%'))
      AND (p_precio_inicio IS NULL OR p.precio >= p_precio_inicio)
      AND (p_precio_fin IS NULL OR p.precio <= p_precio_fin)
      AND (p_tamano IS NULL OR p.`tamaño` = p_tamano)
      AND (
            p_estatus IS NULL
            OR p_estatus = 'todos'
            OR p.estatus = CAST(p_estatus AS UNSIGNED)
          )
      AND (
            p_estatus_receta IS NULL
            OR p_estatus_receta = 'todos'
            OR (p_estatus_receta = '1' AND EXISTS (SELECT 1 FROM recetas r2 WHERE r2.idProducto = p.idProducto))
            OR (p_estatus_receta = '0' AND NOT EXISTS (SELECT 1 FROM recetas r3 WHERE r3.idProducto = p.idProducto))
          )
    ORDER BY p.nombre ASC;
END$$

DELIMITER ;
